git init - אתחול ריפוזיטורי גיט
git add . - הוספת קבצים שעברו עריכה לפני שמירה
git commit -m "commit message" - שמירת העבודה עם הסבר מה נשתנה

git branch -a - תצוגת כל הברנצ'ים בפרוייקט
git checkout main - מעבר לברנץ' - אפשר להשתמש בכל שם אחר חוץ מmain

דחיפה של שינויים לגיטהאב
git push -u origin BRANCH_NAME - דחיפה של הברנץ לשרת בפעם הראשונה
git push - דחיפה של כל השינויים לשרת
